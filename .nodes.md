# build the container

```
docker build -t agent-platform-python-server:latest .
docker build --platform linux/amd64 -t agent-platform-python-server:latest 
```

# load the image into the registry

```
kind load docker-image agent-platform-python-server:latest --name agent-platform
```


# apply the CRD

```
kubectl apply -f config/crd/bases/agentplatform.example.com_multiagentflows.yaml
```

# apply the CR

```
kubectl apply -f config/samples/multiagentflow-sample.yaml
```

# gets the agent pods

```
kubectl get pods -l app=agent-worker
```


# force new pods

```
kubectl delete pod -l app=agent-worker
```
